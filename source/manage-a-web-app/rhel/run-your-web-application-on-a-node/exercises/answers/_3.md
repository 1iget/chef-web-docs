To delete the SQL script, use the `file` resource. Make sure you delete the file _after_ the database is completely initialized, like this.

```ruby
# ~/learn-chef/cookbooks/awesome_customers_rhel/recipes/database.rb
# [...]

# Write schema seed file to filesystem.
cookbook_file node['awesome_customers_rhel']['database']['create_tables_script'] do
  source 'create-tables.sql'
  owner 'root'
  group 'root'
  mode '0600'
end

# Seed the database with a table and test data.
execute "initialize #{node['awesome_customers_rhel']['database']['dbname']} database" do
  command "mysql -h #{node['awesome_customers_rhel']['database']['host']} -u #{node['awesome_customers_rhel']['database']['admin_username']} -p#{node['awesome_customers_rhel']['database']['admin_password']} -D #{node['awesome_customers_rhel']['database']['dbname']} < #{node['awesome_customers_rhel']['database']['create_tables_script']}"
  not_if  "mysql -h #{node['awesome_customers_rhel']['database']['host']} -u #{node['awesome_customers_rhel']['database']['admin_username']} -p#{node['awesome_customers_rhel']['database']['admin_password']} -D #{node['awesome_customers_rhel']['database']['dbname']} -e 'describe customers;'"
end

# Delete the SQL script.
file node['awesome_customers_rhel']['database']['create_tables_script'] do
  action :delete
end
```
